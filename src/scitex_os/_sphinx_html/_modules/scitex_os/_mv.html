def mv(src, tgt):
    import os
    import shutil

    successful = True
    os.makedirs(tgt, exist_ok=True)

    try:
        shutil.move(src, tgt)
        print(f"\nMoved from {src} to {tgt}")
    except OSError as e:
        print(f"\nError: {e}")
        successful = False

    return successful
